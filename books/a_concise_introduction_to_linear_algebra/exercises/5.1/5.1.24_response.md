X = [1; 2; -1; 5; 7];
Y = [2; 2; 0; -2; 0];
Z = [3; 4; 3; 2; -1];
% Model z = c1*x + c2*y + c3
A = [X, Y, ones(size(X))];
s = A \ Z;
% s contains [c1; c2; c3]
% Plotting
plot3(X, Y, Z, 'o'); hold on;
[xx, yy] = meshgrid(min(X):0.5:max(X), min(Y):0.5:max(Y));
zz = s(1)*xx + s(2)*yy + s(3);
surf(xx, yy, zz);
