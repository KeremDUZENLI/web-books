**Rectangular Area Light:**

Point3 sampleRectangle(float width, float height, Point3 center) {
    float u = random(); // [0,1]
    float v = random(); // [0,1]
    float x = (u - 0.5) * width;
    float y = (v - 0.5) * height;
    return center + Vector3(x, y, 0); // Assuming aligned with Z=0
}

Point3 sampleSphere(float radius, Point3 center) {
    float z = 1.0 - 2.0 * random(); // z in [-1, 1]
    float phi = 2.0 * PI * random(); // phi in [0, 2pi]
    
    float r_xy = sqrt(1.0 - z*z);
    float x = r_xy * cos(phi);
    float y = r_xy * sin(phi);
    
    return center + Vector3(x, y, z) * radius;
}
