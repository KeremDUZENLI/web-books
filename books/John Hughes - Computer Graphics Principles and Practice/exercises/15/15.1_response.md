### Answer 15.1
    
To generalize `Image` and `DepthBuffer`, create a templated class `Buffer2D<T>`:

```cpp
template <typename T>
class Buffer2D {
private:
    int m_width;
    int m_height;
    std::vector<T> m_data;

public:
    Buffer2D(int width, int height, const T& initialValue) 
        : m_width(width), m_height(height), m_data(width * height, initialValue) {}

    int width() const { return m_width; }
    int height() const { return m_height; }

    void set(int x, int y, const T& value) {
        if (x >= 0 && x < m_width && y >= 0 && y < m_height) {
            m_data[x + y * m_width] = value;
        }
    }

    const T& get(int x, int y) const {
        return m_data[x + y * m_width];
    }
    
    // Additional methods (save, clear) would be implemented here
};

// Aliases
typedef Buffer2D<Radiance3> Image;
typedef Buffer2D<float> DepthBuffer;
```
