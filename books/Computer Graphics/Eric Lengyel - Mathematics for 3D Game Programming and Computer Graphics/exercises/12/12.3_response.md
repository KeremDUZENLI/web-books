import math
import numpy as np

def check_sphere_collision(p1_initial, v1, r1, p2_initial, v2, r2, time_interval):
    t_start, t_end = time_interval

    # Relative initial position and velocity
    p_rel = p2_initial - p1_initial
    v_rel = v2 - v1

    # Combined radius squared
    r_comb_sq = (r1 + r2) ** 2

    # Quadratic equation coefficients: at^2 + bt + c = 0
    a = np.dot(v_rel, v_rel)
    b = 2 * np.dot(p_rel, v_rel)
    c = np.dot(p_rel, p_rel) - r_comb_sq

    # Check if they are already colliding at t=0
    if c <= 0:
        # Calculate contact point at t=0
        direction = p_rel / np.linalg.norm(p_rel)
        contact_point = p1_initial + direction * r1
        if 0 >= t_start and 0 <= t_end:
             return True, contact_point

    # If a is very small, relative velocity is ~0 (parallel motion)
    if a < 1e-8:
        return False, None

    # Discriminant
    discriminant = b**2 - 4*a*c

    if discriminant < 0:
        return False, None # No real roots, no collision

    # Roots
    sqrt_d = math.sqrt(discriminant)
    t1 = (-b - sqrt_d) / (2*a)
    t2 = (-b + sqrt_d) / (2*a)

    # The first point of contact is the smaller positive root
    t_collision = min(t1, t2)

    if t_start <= t_collision <= t_end:
        # Calculate positions at collision time
        p1_col = p1_initial + v1 * t_collision
        p2_col = p2_initial + v2 * t_collision

        # Point of contact is on the line connecting the centers, r1 away from p1
        direction = (p2_col - p1_col) / np.linalg.norm(p2_col - p1_col)
        contact_point = p1_col + direction * r1

        return True, contact_point

    return False, None
